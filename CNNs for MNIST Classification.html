#reshape data to fit model
#size of each image is 28x28 (we know this from above), and there are 60000 images in
#the training data and 10000 in the test data.
#the 1 at the end indicates that the images are greyscale.
X_train = X_train.reshape(60000,28,28,1)

y_test = to_categorical(y_test)
